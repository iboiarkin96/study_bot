 43    def create(self, payload: UserCreateRequest) -> User:
 44        """Persist a new user or reject a duplicate composite key.
 45
 46        Args:
 47            payload: Validated create body; uniqueness is ``(system_user_id, system_uuid)``.
 48
 49        Returns:
 50            Persisted :class:`~app.models.core.user.User` after save.
 51
 52        Raises:
 53            fastapi.HTTPException: 400 with ``USER_CREATE_ALREADY_EXISTS`` when the composite key exists.
 54        """
 55        su_id = str(payload.system_user_id)
 56        sys_uuid = str(payload.system_uuid)
 57
 58        if self.repository.get_by_system_user_id_and_system_uuid(su_id, sys_uuid) is not None:
 59            logger.warning(
 60                "create_user_duplicate system_user_id=%s system_uuid=%s",
 61                su_id,
 62                sys_uuid,
 63            )
 64            raise HTTPException(
 65                status_code=400,
 66                detail=USER_101.as_detail("business"),
 67            )
 68
 69        user = User(
 70            system_user_id=su_id,
 71            system_uuid=sys_uuid,
 72            username=payload.username,
 73            full_name=payload.full_name,
 74            timezone=payload.timezone,
 75            invalidation_reason_uuid=self._uuid_to_str(payload.invalidation_reason_uuid),
 76            is_row_invalid=payload.is_row_invalid,
 77        )
 78        persisted_user = self.repository.save(user)
 79        logger.info(
 80            "create_user_persisted system_user_id=%s system_uuid=%s client_uuid=%s",
 81            persisted_user.system_user_id,
 82            persisted_user.system_uuid,
 83            persisted_user.client_uuid,
 84        )
 85        return persisted_user

150            )
151        if "username" in data:
152            user.username = cast(Any, data["username"])
153        if "full_name" in data and data["full_name"] is not None:
154            user.full_name = data["full_name"]
155        if "timezone" in data and data["timezone"] is not None:
156            user.timezone = data["timezone"]
157        if "invalidation_reason_uuid" in data:
158            user.invalidation_reason_uuid = cast(
159                Any, self._uuid_to_str(data["invalidation_reason_uuid"])
160            )
161        if "is_row_invalid" in data and data["is_row_invalid"] is not None:
162            user.is_row_invalid = data["is_row_invalid"]
163        if "system_uuid" in data:
164            su = self._uuid_to_str(data["system_uuid"])
165            if su is not None:
166                user.system_uuid = cast(Any, su)
167        return self.repository.save(user)
